#define CONSTANTVALUE 9
if(x==true)
{
  QList< QList< QStringList > > multiStringList;
  callSomeOtherFunction();
  switch(someVar)
  {
    default:
    case CONSTANTVALUE:
    {
      for(int i=0;i<CONSTANTVALUE;i++)
      {
        qDebug()<<"Text: "<< i;
      }
    }
      break;
  }
}
