#BASEADO EM http://doc.qt.nokia.com/qtmobility-1.0-tp/installation.html#building-on-windows

cd
cd\
mkdir qt
c:\qt\
#FAZENDO DOWNLOAD SOURCE ATUALIZADO
git clone git://gitorious.org/qt-mobility/qt-mobility.git
#aqui vai demorar uns minutos dependendo da sua vecidade de download

#REMOVENDO PATHS DESNECESSARIOS
cd \
set path=
#INDICADO O LOCA O COMPILADOR, O QMAKE QUE DESEJA UTILIZAR, E PERL
set path=c:\mingw\bin;c:\qt\4.7.4w32\bin;c:\Perl\bin;
ou
set path=c:\mingw\bin;c:\qt\4.7.4w32\bin;c:\Perl64\bin;

#ENTRANDO NO DIRETORIO
cd c:\qt\
cd c:\qt\qt-mobility

#as vezes as versoes nao compilao entao vamos escolher uma taf especifica
#no meu caso a versao atual é a [v1.2.0-tp1] e esta tendo problema com o SDK do windows
git tag -l

#apos escolher a tag digita os seguintes comandos #remove alteracoes feitas na pasta
git stash save #retorna para um tag de sua escolha
git checkout v1.1.3

#CONFIGURANDO PARA O BUILD NO LOCAL DO SDK QUE FOI INDICADO
#aqui vai demorar uns minutos dependendo do seu CPU
configure -prefix C:\qt\4.7.4w32

#EXECUTANDO BUILD
#aqui vai demorar uns minutos dependendo do seu CPU
mingw32-make.exe

#Após o build é necessario instalar no SDK definido no configure
mingw32-make.exe install

JA NO PROJETO É NECESSARIO INCLUIR:
CONFIG += mobility
MOBILITY = systeminfo
